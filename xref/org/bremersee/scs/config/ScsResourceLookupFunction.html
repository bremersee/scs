17  package org.bremersee.scs.config;
18  
19  import static org.bremersee.scs.config.ScsProperties.DEFAULT_CLASSPATH_LOCATION;
20  import static org.bremersee.scs.config.ScsProperties.createContentResource;
21  
22  import java.io.File;
23  import java.io.IOException;
24  import java.io.UncheckedIOException;
25  import java.nio.charset.StandardCharsets;
26  import java.util.function.Function;
27  import javax.validation.Valid;
28  import javax.validation.constraints.NotNull;
29  import lombok.extern.slf4j.Slf4j;
30  import org.springframework.core.io.Resource;
31  import org.springframework.http.server.PathContainer;
32  import org.springframework.util.StringUtils;
33  import org.springframework.validation.annotation.Validated;
34  import org.springframework.web.reactive.function.server.ServerRequest;
35  import org.springframework.web.util.pattern.PathPattern;
36  import org.springframework.web.util.pattern.PathPatternParser;
37  import reactor.core.publisher.Mono;
38  
39  /**
40   * The static content resource lookup function.
41   *
42   * @author Christian Bremer
43   */
44  @Validated
45  @Slf4j
46  public class ScsResourceLookupFunction implements Function<ServerRequest, Mono<Resource>> {
47  
48    private static final PathPatternParser PATTERN_PARSER = new PathPatternParser();
49  
50    private final PathPattern pattern;
51  
52    private final Resource location;
53  
54    private final ScsProperties properties;
55  
56    /**
57     * Instantiates a new resource lookup function.
58     *
59     * @param properties the properties
60     */
61    public ScsResourceLookupFunction(@NotNull @Valid ScsProperties properties) {
62      this.properties = properties;
63      this.pattern = PATTERN_PARSER.parse(properties.getPattern());
64      if (!this.properties.getContentLocation().toLowerCase()
65          .startsWith(ScsProperties.CLASSPATH_PREFIX)) {
66        File dir = new File(this.properties.getContentLocation());
67        if (!dir.exists() || !dir.isDirectory()) {
68          log.warn("Content location {} does not exist, using fallback location {}",
69              this.properties.getContentLocation(), DEFAULT_CLASSPATH_LOCATION);
70          this.location = createContentResource(DEFAULT_CLASSPATH_LOCATION);
71        } else {
72          this.location = properties.createContentResource();
73        }
74      } else {
75        this.location = properties.createContentResource();
76      }
77    }
78  
79    @SuppressWarnings("BlockingMethodInNonBlockingContext")
80    @Override
81    public Mono<Resource> apply(ServerRequest request) {
82      PathContainer pathContainer = request.pathContainer();
83      log.debug("Looking for static content of request path {}", pathContainer.value());
84      if (!this.pattern.matches(pathContainer)) {
85        log.debug("Pattern {} does not match request path, return no static content resource.",
86            properties.getPattern());
87        return Mono.empty();
88      }
89      pathContainer = this.pattern.extractPathWithinPattern(pathContainer);
90      String path = pathContainer.value();
91      log.debug("Extracted path is {}", path);
92      path = processPath(pathContainer.value());
93      log.debug("Processed path is {}", path);
94      if (path.contains("%")) {
95        path = StringUtils.uriDecode(path, StandardCharsets.UTF_8);
96        log.debug("Url decoded path is {}", path);
97      }
98      if (!StringUtils.hasLength(path)) {
99        log.debug("Path is empty, trying to find an index file.");
100       path = findDirectoryIndex(path);
101       if (StringUtils.isEmpty(path)) {
102         log.debug("No index file was found, return no static content resource.");
103         return Mono.empty();
104       }
105       log.debug("An index file was found, path is {}", path);
106     }
107 
108     try {
109       Resource resource = this.location.createRelative(path);
110       boolean exists = resource.exists();
111       boolean isReadable = resource.isReadable();
112       log.debug("Resource exists? {}, resource is readable? {}", exists, isReadable);
113       if (exists && !isReadable) {
114         String pathWithIndex = findDirectoryIndex(path);
115         if (path.equals(pathWithIndex)) {
116           log.debug("No index file was found, return no static content resource.");
117           return Mono.empty();
118         }
119         path = pathWithIndex;
120         resource = this.location.createRelative(path);
121         isReadable = resource.isReadable();
122       }
123       if (exists && isReadable) {
124         log.debug("Returning static content resource {}", path);
125         return Mono.just(resource);
126       } else {
127         log.debug("Resource ('{}') does not exists.", path);
128         return Mono.empty();
129       }
130     } catch (IOException ex) {
131       throw new UncheckedIOException(ex);
132     }
133   }
134 
135   private String processPath(String path) {
136     boolean slash = false;
137     for (int i = 0; i < path.length(); i++) {
138       if (path.charAt(i) == '/') {
139         slash = true;
140       } else if (path.charAt(i) > ' ' && path.charAt(i) != 127) {
141         if (i == 0 || (i == 1 && slash)) {
142           return path;
143         }
144         path = slash ? "/" + path.substring(i) : path.substring(i);
145         return path;
146       }
147     }
148     return (slash ? "/" : "");
149   }
150 
151   private String findDirectoryIndex(String path) {
152     return properties.findDirectoryIndex(path)
153         .map(index -> {
154           if (StringUtils.hasLength(path)) {
155             return path + "/" + index;
156           }
157           return index;
158         })
159         .orElse(path);
160   }
