20  import java.util.Map;
21  import java.util.Optional;
22  import javax.validation.constraints.NotNull;
23  import lombok.EqualsAndHashCode;
24  import lombok.Getter;
25  import lombok.NoArgsConstructor;
26  import lombok.Setter;
27  import lombok.ToString;
28  import lombok.extern.slf4j.Slf4j;
29  import org.springframework.boot.context.properties.ConfigurationProperties;
30  import org.springframework.cloud.context.config.annotation.RefreshScope;
31  import org.springframework.core.io.ClassPathResource;
32  import org.springframework.core.io.FileSystemResource;
33  import org.springframework.core.io.Resource;
34  import org.springframework.data.util.Pair;
35  import org.springframework.util.AntPathMatcher;
36  import org.springframework.validation.annotation.Validated;
37  
38  /**
39   * The static content server properties.
40   *
41   * @author Christian Bremer
42   */
43  @ConfigurationProperties(prefix = "bremersee.scs")
44  @RefreshScope
45  @Getter
46  @Setter
47  @ToString
48  @EqualsAndHashCode
49  @NoArgsConstructor
50  @Validated
51  @Slf4j
52  public class ScsProperties {
53  
54    /**
55     * The classpath prefix.
56     */
57    static final String CLASSPATH_PREFIX = "classpath:";
58  
59    /**
60     * The default classpath location.
61     */
62    static final String DEFAULT_CLASSPATH_LOCATION = CLASSPATH_PREFIX + "content/";
63  
64    @NotNull
65    private String pattern = "/**";
66  
67    @NotNull
68    private String contentLocation = "/opt/content/";
69  
70    private String defaultIndex = "index.html";
71  
72    @NotNull
73    private Map<String, String> directoryPatternIndexMap = new LinkedHashMap<>();
74  
75    /**
76     * Gets content location.
77     *
78     * @return the content location
79     */
80    public String getContentLocation() {
81      return Optional.of(contentLocation)
82          .map(String::trim)
83          .map(value -> !value.endsWith("/") ? value + "/" : value)
84          .orElse(contentLocation);
85    }
86  
87    /**
88     * Create content root resource.
89     *
90     * @return the resource
91     */
92    public Resource createContentResource() {
93      return createContentResource(getContentLocation());
94    }
95  
96    /**
97     * Create content root resource.
98     *
99     * @param location the location
100    * @return the resource
101    */
102   static Resource createContentResource(@NotNull String location) {
103     return location.toLowerCase().startsWith(CLASSPATH_PREFIX)
104         ? new ClassPathResource(location.substring(CLASSPATH_PREFIX.length()))
105         : new FileSystemResource(location);
106   }
107 
108   /**
109    * Find directory index.
110    *
111    * @param requestPath the request path
112    * @return the directory index
113    */
114   public Optional<String> findDirectoryIndex(String requestPath) {
115     final String path;
116     if (requestPath != null) {
117       String tmp = requestPath.trim();
118       path = tmp.startsWith("/") ? tmp : "/" + tmp;
119     } else {
120       path = "/";
121     }
122     log.debug("Looking for index of {}", path);
123     Map<String, String> map = getDirectoryPatternIndexMap();
124     if (!(map.containsKey("/**") || map.containsKey("[/**]"))) {
125       map.put("/**", defaultIndex);
126     }
127     return map.entrySet().stream()
128         .map(entry -> {
129           String key = entry.getKey();
130           log.debug("Key {} -> {}", entry.getKey(), key);
131           return Pair.of(key, entry.getValue());
132         })
133         .peek(pair -> log.debug("Pattern {} matches path {}? {}", pair.getFirst(), path,
134             new AntPathMatcher().match(pair.getFirst(), path)))
135         .filter(pair -> new AntPathMatcher().match(pair.getFirst(), path))
136         .findFirst()
137         .map(Pair::getSecond)
138         .map(index -> {
139           if (index.startsWith("/")) {
140             return index.substring(1);
141           }
142           return index;
143         });
144   }
145 
146 }
